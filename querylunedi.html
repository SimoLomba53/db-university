QUERY 1

QUERY 2

SELECT `degrees`.*,`departments`.* FROM `degrees` JOIN `departments` ON `department_id`=`degrees`.`id` WHERE `degrees`.`level`= "magistrale" AND `departments`.`name`=" Dipartimento di Neuroscienze"

QUERY 3
